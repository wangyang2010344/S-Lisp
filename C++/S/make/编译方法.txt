//设置被转的文件尾缀
var white_suffixs=[
	".h",
	".lisp",
	".txt",
	".js",
	".cpp"
];
module.exports={
	gitPath:"",//UTF-8文件目录，即git项目目录
	vsPath:"",//生成供vs调用的C++文件目录
    ...
}
